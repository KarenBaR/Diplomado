-- ═══════════════════════════════════════════════════════════════════════════
-- PASO 1: CREAR TABLA CON PARTICIÓN POR CODMES
-- ═══════════════════════════════════════════════════════════════════════════
--DROP TABLE disc_comercial.HM_UNIVERSO_VINCULADOS_BPE


CREATE TABLE disc_comercial.HM_UNIVERSO_VINCULADOS_BPE
WITH (
    format = 'Parquet',
    parquet_compression = 'SNAPPY',
    partitioned_by = ARRAY['codmes']
)
AS (

-- ═══════════════════════════════════════════════════════════════════════════
-- PASO 2: UNIVERSO BASE (CLIENTES TIENDAS)
-- ═══════════════════════════════════════════════════════════════════════════

WITH funnel AS (
    SELECT
        numeroruc        AS num_ruc,
        periodo          AS periodo_campania,
        canal,
        campanha,
        supervisor,
        ejecutivo,
        flg_gestionado,
        flg_acepta1,
        flg_ce,
        flg_desembolsado,
        colocacion
    FROM disc_comercial.HM_FUNNEL_NEGOCIO_ACTUAL_VF f
    WHERE --campanha <> 'SIN CAMPAÑA' AND
         fecha_carga = (SELECT MAX(fecha_carga)
            FROM disc_comercial.HM_FUNNEL_NEGOCIO_ACTUAL_VF
            WHERE periodo = f.periodo)
),

-- ═══════════════════════════════════════════════════════════════════════════
-- PASO 3: REPRESENTANTES LEGALES (HASH → RUC EMPRESA)
-- ═══════════════════════════════════════════════════════════════════════════

rep_legal AS (
    SELECT
        ruc_value       AS ruc_hash_empresa,
        tip_doc,
        key_value       AS dni_hash_rrll,
        des_cargo,
        fec_vigencia,
        estado,
        fec_proceso,
        est_registro
    FROM e_perm_aws.t_mst_rep_legal_sunat_rsk  WHERE estado = '0'
),

-- ═══════════════════════════════════════════════════════════════════════════
-- PASO 4: DATOS RENIEC (GÉNERO + CÓNYUGE)
-- ═══════════════════════════════════════════════════════════════════════════

reniec AS (
    SELECT
        tip_doc,
        key_value       AS dni_hash,
        fec_nacimiento,
        est_civil,
        genero,
        tipdoc_conyugue,
        coddoc_conyugue AS dni_hash_conyugue,
        gen_conyugue,
        fec_dato
    FROM e_perm_aws.t_mst_reniec_rsk
    WHERE 
        fec_dato = (
            SELECT MAX(fec_dato)
            FROM e_perm_aws.t_mst_reniec_rsk
        )
),

-- ═══════════════════════════════════════════════════════════════════════════
-- PASO 5: RCC PERSONAS NATURALES (DEUDA + CALIFICACIÓN)
-- ═══════════════════════════════════════════════════════════════════════════

rcc_personas AS (
    SELECT
        codmes,
        tip_doc,
        key_value       AS dni_hash,
        
        -- ── M01 (mes más reciente) ────────────────────────
        lintot_m01,
        saltot_m01,
        salvig_m01,
        clasif_sbs_m01,
        pct_clasif_per_m01,
        pct_clasif_dud_m01,
        pct_clasif_def_m01,
        pct_clasif_cpp_m01,
        pct_clasif_nor_m01,
        
        -- ── M02 (mes anterior) ────────────────────────────
        saltot_m02,
        salvig_m02,
        clasif_sbs_m02,
        
        -- ── M03 (2 meses antes) ───────────────────────────
        saltot_m03,
        salvig_m03,
        clasif_sbs_m03,
        
        -- ── Agregados útiles ──────────────────────────────
        ctd_ent_tot_m01 AS cant_entidades,
        ctd_prod_m01    AS cant_productos
        
    FROM e_perm_aws.tbl_rcc_per_allsf_mdl
),
-- ═══════════════════════════════════════════════════════════════════════════
-- PASO 6: CALCULADORA SSFF
-- ═══════════════════════════════════════════════════════════════════════════

calculadora_ssff AS (
   SELECT
    codmes_ejecucion,
    key_value,
    /* ── NEGOCIO: todo lo que empieza con Ng_ ───────────────────── */
    SUM(CASE
        WHEN producto_dsc LIKE 'Ng!_%' ESCAPE '!'
        THEN cuota_val ELSE 0
    END) AS cuota_negocio,
    /* ── CONSUMO: C_* (excepto préstamo personal) + Credito consumo* + NR-TC */
    SUM(CASE
        WHEN (producto_dsc LIKE 'C!_%' ESCAPE '!'
              AND NOT (UPPER(producto_dsc) = 'C_LINEAUTILIZADA'
                       AND LOWER(subproducto_dsc) LIKE '%prestamo%'))
          OR producto_dsc LIKE 'Credito consumo%'
          OR producto_dsc = 'Saldo Credito NR-TC'
        THEN cuota_val ELSE 0
    END) AS cuota_consumo,
 
    /* ── PRÉSTAMO PERSONAL: C_LineaUtilizada + subproducto prestamo */
    SUM(CASE
        WHEN UPPER(producto_dsc) = 'C_LINEAUTILIZADA'
         AND LOWER(subproducto_dsc) LIKE '%prestamo%'
        THEN cuota_val ELSE 0
    END) AS cuota_prestamo_personal,
 
    /* ── HIPOTECARIO ─────────────────────────────────────────────── */
    SUM(CASE
        WHEN producto_dsc = 'Credito Hipotecario'
        THEN cuota_val ELSE 0
    END) AS cuota_hipotecario,
    /* ── TOTAL ───────────────────────────────────────────────────── */
    SUM(cuota_val) AS cuota_total

FROM e_perm_aws.t_agg_cbpe_calculadora_ssff 
WHERE key_value IS NOT NULL AND key_value <> ''
GROUP BY
    codmes_ejecucion,
    key_value

),
-- ═══════════════════════════════════════════════════════════════════════════
-- PASO 7: JOIN FUNNEL → REP LEGAL → RENIEC → RCC
-- ═══════════════════════════════════════════════════════════════════════════

base_con_rrll AS (
    SELECT
        -- ── Datos cliente empresa ─────────────────────────
        f.num_ruc,
        f.periodo_campania,
        f.canal,
        f.campanha,
        f.supervisor,
        f.ejecutivo,
        f.flg_gestionado,
        f.flg_acepta1,
        f.flg_ce,
        f.flg_desembolsado,
        f.colocacion,
        
        -- ── Datos representante legal ─────────────────────
        CASE WHEN rl.dni_hash_rrll IS NOT NULL THEN rl.dni_hash_rrll ELSE D.key_value END dni_hash_rrll,
        rl.des_cargo,
        rl.fec_vigencia,
        
        -- ── Datos RENIEC representante ────────────────────
        r.fec_nacimiento        AS rrll_fec_nacimiento,
        r.est_civil             AS rrll_est_civil,
        r.genero                AS rrll_genero,
        r.dni_hash_conyugue     AS rrll_dni_conyugue,
        r.gen_conyugue          AS rrll_gen_conyugue,
        
        -- ── RCC representante legal ───────────────────────
        rcc.lintot_m01          AS rrll_linea_total,
        rcc.saltot_m01          AS rrll_deuda_total,
        rcc.salvig_m01          AS rrll_deuda_vigente,
        rcc.clasif_sbs_m01      AS rrll_calificacion,
        rcc.pct_clasif_per_m01  AS rrll_pct_perdida,
        rcc.pct_clasif_dud_m01  AS rrll_pct_dudoso,
        rcc.pct_clasif_def_m01  AS rrll_pct_deficiente,
        rcc.pct_clasif_cpp_m01  AS rrll_pct_cpp,
        rcc.pct_clasif_nor_m01  AS rrll_pct_normal,
        rcc.cant_entidades      AS rrll_cant_entidades,
        rcc.cant_productos      AS rrll_cant_productos,
        
        -- ── Calificación dominante RRLL ───────────────────
        CASE
            WHEN COALESCE(rcc.pct_clasif_per_m01, 0) > 0 THEN 'PER'
            WHEN COALESCE(rcc.pct_clasif_dud_m01, 0) > 0 THEN 'DUD'
            WHEN COALESCE(rcc.pct_clasif_def_m01, 0) > 0 THEN 'DEF'
            WHEN COALESCE(rcc.pct_clasif_cpp_m01, 0) > 0 THEN 'CPP'
            WHEN COALESCE(rcc.pct_clasif_nor_m01, 0) > 0 THEN 'NOR'
            ELSE 'SIN_DEUDA'
        END AS rrll_calif_dominante,
        
        -- ── Variaciones deuda RRLL (m01 vs m03) ──────────
        COALESCE(rcc.saltot_m01, 0) - COALESCE(rcc.saltot_m03, 0) AS rrll_var_deuda_3m,
        COALESCE(cale.cuota_negocio,0) cuota_negocio_empresa,
        COALESCE(cale.cuota_consumo,0) cuota_consumo_empresa,
        COALESCE(cale.cuota_prestamo_personal,0) cuota_prestamo_personal_empresa,
        COALESCE(cale.cuota_hipotecario,0) cuota_hipotecario_empresa,
        COALESCE(cale.cuota_total,0) cuota_total_empresa,
       
        COALESCE(calr.cuota_negocio,0) cuota_negocio_rrll,
        COALESCE(calr.cuota_consumo,0) cuota_consumo_rrll,
        COALESCE(calr.cuota_prestamo_personal,0) cuota_prestamo_personal_rrll,
        COALESCE(calr.cuota_hipotecario,0) cuota_hipotecario_rrll,
        COALESCE(calr.cuota_total,0) cuota_total_rrll
        
    FROM funnel f
    
    --- RUC TO DNI 
     LEFT JOIN e_perm_aws.V_MST_ENCRIP_DNI_RUC_PN_HIST D
        ON f.num_ruc = D.key_value_ruc_pn

    -- Join representantes legales
    LEFT JOIN rep_legal rl
        ON f.num_ruc = rl.ruc_hash_empresa
    
    -- Join RENIEC
    LEFT JOIN reniec r
        ON rl.dni_hash_rrll = r.dni_hash
        AND rl.tip_doc = r.tip_doc
    
    -- Join RCC personas (periodo - 2 meses)
    LEFT JOIN rcc_personas rcc
        ON rl.dni_hash_rrll = rcc.dni_hash
        AND rl.tip_doc = rcc.tip_doc
        AND rcc.codmes = date_format(date_add('month', -2, date_parse(f.periodo_campania, '%Y%m')), '%Y%m' )
      
    -- Join Calculadora SSFF BPE 
    LEFT JOIN calculadora_ssff cale
        ON f.num_ruc = cale.key_value
        AND cale.codmes_ejecucion = date_format( date_add('month', -1, date_parse(f.periodo_campania, '%Y%m')),  '%Y%m' ) 
        
    -- Join Calculadora SSFF BPE 
    LEFT JOIN calculadora_ssff calr
        ON rl.dni_hash_rrll = calr.key_value 
        AND calr.codmes_ejecucion = date_format( date_add('month', -1, date_parse(f.periodo_campania, '%Y%m')),  '%Y%m' ) 
),

-- ═══════════════════════════════════════════════════════════════════════════
-- PASO 7: AGREGAR RCC DEL CÓNYUGE (SI EXISTE)
-- ═══════════════════════════════════════════════════════════════════════════

base_con_conyugue AS (
    SELECT
        b.*,
        
        -- ── RCC cónyuge ───────────────────────────────────
        rcc_con.saltot_m01      AS conyugue_deuda_total,
        rcc_con.salvig_m01      AS conyugue_deuda_vigente,
        rcc_con.clasif_sbs_m01  AS conyugue_calificacion,
        rcc_con.pct_clasif_per_m01 AS conyugue_pct_perdida,
        rcc_con.pct_clasif_dud_m01 AS conyugue_pct_dudoso,
        rcc_con.pct_clasif_def_m01 AS conyugue_pct_deficiente,
        rcc_con.pct_clasif_cpp_m01 AS conyugue_pct_cpp,
        rcc_con.pct_clasif_nor_m01 AS conyugue_pct_normal,
        rcc_con.cant_entidades  AS conyugue_cant_entidades,
        
        -- ── Calificación dominante cónyuge ────────────────
        CASE
            WHEN COALESCE(rcc_con.pct_clasif_per_m01, 0) > 0 THEN 'PER'
            WHEN COALESCE(rcc_con.pct_clasif_dud_m01, 0) > 0 THEN 'DUD'
            WHEN COALESCE(rcc_con.pct_clasif_def_m01, 0) > 0 THEN 'DEF'
            WHEN COALESCE(rcc_con.pct_clasif_cpp_m01, 0) > 0 THEN 'CPP'
            WHEN COALESCE(rcc_con.pct_clasif_nor_m01, 0) > 0 THEN 'NOR'
            ELSE 'SIN_DEUDA'
        END AS conyugue_calif_dominante,

        COALESCE(cal.cuota_negocio,0) cuota_negocio_conyugue,
        COALESCE(cal.cuota_consumo,0) cuota_consumo_conyugue,
        COALESCE(cal.cuota_prestamo_personal,0) cuota_prestamo_personal_conyugue,
        COALESCE(cal.cuota_hipotecario,0) cuota_hipotecario_conyugue,
        COALESCE(cal.cuota_total,0) cuota_total_conyugue
        
    FROM base_con_rrll b
    
    -- Join RCC del cónyuge
    LEFT JOIN rcc_personas rcc_con
        ON b.rrll_dni_conyugue = rcc_con.dni_hash
        AND rcc_con.codmes = date_format(date_add('month', -2, date_parse(b.periodo_campania, '%Y%m')),  '%Y%m')
        
    -- Join Calculadora SSFF BPE 
    LEFT JOIN calculadora_ssff cal
        ON b.rrll_dni_conyugue = cal.key_value
        AND cal.codmes_ejecucion = date_format( date_add('month', -1, date_parse(b.periodo_campania, '%Y%m')),  '%Y%m' )      
        
),

-- ═══════════════════════════════════════════════════════════════════════════
-- PASO 8: CALCULAR SEÑALES AGREGADAS POR EMPRESA
-- ═══════════════════════════════════════════════════════════════════════════

senales_vinculados AS (
    SELECT
        num_ruc,
        periodo_campania,
        
        -- ── Count representantes legales ──────────────────
        COUNT(DISTINCT dni_hash_rrll) AS total_rrll,
        
        -- ── Señales de riesgo RRLL ────────────────────────
        SUM(CASE 
            WHEN COALESCE(rrll_pct_perdida, 0) > 0
                 OR COALESCE(rrll_pct_dudoso, 0) > 0
                 OR COALESCE(rrll_pct_deficiente, 0) > 0 THEN 1 
            ELSE 0 
        END) AS rrll_con_mala_calif,
        
        SUM(CASE 
            WHEN COALESCE(rrll_deuda_total, 0) > 0 THEN 1 
            ELSE 0 
        END) AS rrll_con_deuda,
        
        SUM(CASE 
            WHEN COALESCE(rrll_pct_cpp, 0) > 0
                 OR COALESCE(rrll_pct_deficiente, 0) > 0
                 OR COALESCE(rrll_pct_dudoso, 0) > 0
                 OR COALESCE(rrll_pct_perdida, 0) > 0 THEN 1
            ELSE 0
        END) AS rrll_con_calificacion_riesgosa,
        
        -- ── Señales de riesgo CÓNYUGE ─────────────────────
        SUM(CASE 
            WHEN COALESCE(conyugue_pct_perdida, 0) > 0
                 OR COALESCE(conyugue_pct_dudoso, 0) > 0
                 OR COALESCE(conyugue_pct_deficiente, 0) > 0 THEN 1
            ELSE 0
        END) AS conyugue_con_mala_calif,
        
        -- ── Deuda total vinculados ────────────────────────
        SUM(COALESCE(rrll_deuda_total, 0)) AS deuda_total_rrll,
        SUM(COALESCE(conyugue_deuda_total, 0)) AS deuda_total_conyugue,
        SUM(
            COALESCE(rrll_deuda_total, 0) + 
            COALESCE(conyugue_deuda_total, 0)
        ) AS deuda_total_vinculados,
        
        -- ── Peor calificación score ───────────────────────
        MAX(
            CASE 
                WHEN COALESCE(rrll_pct_perdida, 0) > 0 THEN 5
                WHEN COALESCE(rrll_pct_dudoso, 0) > 0 THEN 4
                WHEN COALESCE(rrll_pct_deficiente, 0) > 0 THEN 3
                WHEN COALESCE(rrll_pct_cpp, 0) > 0 THEN 2
                WHEN COALESCE(rrll_pct_normal, 0) > 0 THEN 1
                ELSE 0
            END
        ) AS peor_calificacion_score,
        
        -- ── Flag bloqueo automático ───────────────────────
        MAX(CASE 
            WHEN COALESCE(rrll_pct_cpp, 0) > 0
                 OR COALESCE(rrll_pct_deficiente, 0) > 0
                 OR COALESCE(rrll_pct_dudoso, 0) > 0
                 OR COALESCE(rrll_pct_perdida, 0) > 0
                 OR COALESCE(conyugue_pct_cpp, 0) > 0
                 OR COALESCE(conyugue_pct_deficiente, 0) > 0
                 OR COALESCE(conyugue_pct_dudoso, 0) > 0
                 OR COALESCE(conyugue_pct_perdida, 0) > 0 THEN 1
            ELSE 0
        END) AS flg_bloqueo_vinculados
        
    FROM base_con_conyugue
    GROUP BY num_ruc, periodo_campania
)

-- ═══════════════════════════════════════════════════════════════════════════
-- PASO 9: RESULTADO FINAL CON PARTICIÓN
-- ═══════════════════════════════════════════════════════════════════════════

SELECT
    b.*,
    s.total_rrll,
    s.rrll_con_mala_calif,
    s.rrll_con_deuda,
    s.rrll_con_calificacion_riesgosa,
    s.conyugue_con_mala_calif,
    s.deuda_total_rrll,
    s.deuda_total_conyugue,
    s.deuda_total_vinculados,
    s.peor_calificacion_score,
    s.flg_bloqueo_vinculados,
    
    -- ── Clasificación nivel riesgo vinculados ────────
    CASE
        WHEN s.flg_bloqueo_vinculados = 1 THEN 'ALTO'
        WHEN s.rrll_con_mala_calif >= 1
             OR s.conyugue_con_mala_calif >= 1 THEN 'MEDIO'
        WHEN s.deuda_total_vinculados > 50000 THEN 'MEDIO'
        WHEN s.total_rrll = 0 THEN 'SIN_DATOS'
        ELSE 'BAJO'
    END AS nivel_riesgo_vinculados,

    
    -- ── Campo para partición ──────────────────────────
    date_format(
        date_add('month', -2, date_parse(b.periodo_campania, '%Y%m')), 
        '%Y%m'
    ) AS codmes

FROM base_con_conyugue b

LEFT JOIN senales_vinculados s
    ON b.num_ruc = s.num_ruc
    AND b.periodo_campania = s.periodo_campania

);


SELECT * FROM  disc_comercial.HM_UNIVERSO_VINCULADOS_BPE LIMIT 100

-- ═══════════════════════════════════════════════════════════════════════════
-- PASO 1: CREAR TABLA CON PARTICIÓN POR CODMES
-- ═══════════════════════════════════════════════════════════════════════════
--  DROP TABLE disc_comercial.HM_UNIVERSO_CUOTAS_SSFF


CREATE TABLE disc_comercial.HM_UNIVERSO_CUOTAS_SSFF
WITH (
    format = 'Parquet',
    parquet_compression = 'SNAPPY',
    partitioned_by = ARRAY['codmes']
)
AS (

WITH base AS (
    SELECT
        periodo_campania,
        periodo_campania codmes,
        num_ruc,
        dni_hash_rrll,
        COALESCE(cuota_negocio_empresa, 0) AS cuota_negocio_empresa,
        COALESCE(cuota_consumo_empresa, 0) AS cuota_consumo_empresa,
        COALESCE(cuota_prestamo_personal_empresa, 0) AS cuota_prestamo_personal_empresa,
        COALESCE(cuota_hipotecario_empresa, 0) AS cuota_hipotecario_empresa,
        COALESCE(cuota_total_empresa, 0) AS cuota_total_empresa,
        COALESCE(cuota_negocio_rrll, 0) AS cuota_negocio_rrll,
        COALESCE(cuota_consumo_rrll, 0) AS cuota_consumo_rrll,
        COALESCE(cuota_prestamo_personal_rrll, 0) AS cuota_prestamo_personal_rrll,
        COALESCE(cuota_hipotecario_rrll, 0) AS cuota_hipotecario_rrll,
        COALESCE(cuota_total_rrll, 0) AS cuota_total_rrll,
        COALESCE(cuota_total_conyugue, 0) AS cuota_total_conyugue,
        COALESCE(cuota_negocio_conyugue, 0) AS cuota_negocio_conyugue,
        COALESCE(cuota_consumo_conyugue, 0) AS cuota_consumo_conyugue,
        COALESCE(cuota_prestamo_personal_conyugue, 0) AS cuota_prestamo_personal_conyugue,
        COALESCE(cuota_hipotecario_conyugue, 0) AS cuota_hipotecario_conyugue

    FROM disc_comercial.HM_UNIVERSO_VINCULADOS_BPE
  
),
empresa AS (
    SELECT
        periodo_campania,
        codmes,
        num_ruc,
        MAX(cuota_negocio_empresa) AS cuota_negocio_empresa,
        MAX(cuota_consumo_empresa) AS cuota_consumo_empresa,
        MAX(cuota_prestamo_personal_empresa) AS cuota_prestamo_personal_empresa,
        MAX(cuota_hipotecario_empresa) AS cuota_hipotecario_empresa,
        MAX(cuota_total_empresa) AS cuota_total_empresa
    FROM base
    GROUP BY 1, 2,3
),
rrll_rep AS (
    SELECT
        periodo_campania,
        num_ruc,
        dni_hash_rrll,
        MAX(cuota_negocio_rrll) AS cuota_negocio_rrll,
        MAX(cuota_consumo_rrll) AS cuota_consumo_rrll,
        MAX(cuota_prestamo_personal_rrll) AS cuota_prestamo_personal_rrll,
        MAX(cuota_hipotecario_rrll) AS cuota_hipotecario_rrll,
        MAX(cuota_total_rrll) AS cuota_total_rrll,
        MAX(cuota_total_conyugue) AS cuota_total_conyugue,
        MAX(cuota_negocio_conyugue) AS cuota_negocio_conyugue,
        MAX(cuota_consumo_conyugue) AS cuota_consumo_conyugue,
        MAX(cuota_prestamo_personal_conyugue) AS cuota_prestamo_personal_conyugue,
        MAX(cuota_hipotecario_conyugue) AS cuota_hipotecario_conyugue

    FROM base
    GROUP BY 1, 2, 3
),

rrll AS (
    SELECT
        periodo_campania,
        num_ruc,
        SUM(cuota_negocio_rrll) AS cuota_negocio_rrll,
        SUM(cuota_consumo_rrll) AS cuota_consumo_rrll,
        SUM(cuota_prestamo_personal_rrll) AS cuota_prestamo_personal_rrll,
        SUM(cuota_hipotecario_rrll) AS cuota_hipotecario_rrll,
        SUM(cuota_total_rrll) AS cuota_total_rrll,
        SUM(cuota_total_conyugue) AS cuota_total_conyugue,
        SUM(cuota_negocio_conyugue) AS cuota_negocio_conyugue,
        SUM(cuota_consumo_conyugue) AS cuota_consumo_conyugue,
        SUM(cuota_prestamo_personal_conyugue) AS cuota_prestamo_personal_conyugue,
        SUM(cuota_hipotecario_conyugue) AS cuota_hipotecario_conyugue,

        COUNT(DISTINCT dni_hash_rrll) AS nro_rrll_unicos
    FROM rrll_rep
    GROUP BY 1, 2
)
SELECT
    e.periodo_campania,
    e.num_ruc,
    e.cuota_negocio_empresa,
    e.cuota_consumo_empresa,
    e.cuota_prestamo_personal_empresa,
    e.cuota_hipotecario_empresa,
    e.cuota_total_empresa,
    r.cuota_negocio_rrll,
    r.cuota_consumo_rrll,
    r.cuota_prestamo_personal_rrll,
    r.cuota_hipotecario_rrll,
    r.cuota_total_rrll,
    COALESCE(e.cuota_total_empresa, 0) + COALESCE(r.cuota_total_rrll, 0) AS cuota_sf,
    COALESCE(e.cuota_total_empresa, 0) + COALESCE(r.cuota_total_rrll, 0) +  COALESCE(r.cuota_total_conyugue, 0) AS cuota_sf_2,
    r.cuota_total_conyugue,
    r.cuota_negocio_conyugue,
    r.cuota_consumo_conyugue,
    r.cuota_prestamo_personal_conyugue,
    r.cuota_hipotecario_conyugue,
    r.nro_rrll_unicos,
    codmes
FROM empresa e
LEFT JOIN rrll r
    ON e.periodo_campania = r.periodo_campania
   AND e.num_ruc = r.num_ruc

ORDER BY 1, 2
)
